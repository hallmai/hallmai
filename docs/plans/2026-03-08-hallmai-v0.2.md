```typescript
// backend/src/modules/conversation/conversation.service.ts
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Between, Repository } from 'typeorm';
import { Conversation } from '../../common/entity/conversation.entity';

@Injectable()
export class ConversationService {
  constructor(
    @InjectRepository(Conversation)
    private readonly conversationRepository: Repository<Conversation>,
  ) {}

  async create(deviceId: number): Promise<Conversation> {
    const conversation = this.conversationRepository.create({
      deviceId,
      startedAt: new Date(),
    });
    return this.conversationRepository.save(conversation);
  }

  async end(
    id: number,
    transcript: string,
  ): Promise<Conversation> {
    const conversation = await this.conversationRepository.findOneByOrFail({ id });
    conversation.endedAt = new Date();
    conversation.durationSeconds = Math.floor(
      (conversation.endedAt.getTime() - conversation.startedAt.getTime()) / 1000,
    );
    conversation.transcript = transcript;
    return this.conversationRepository.save(conversation);
  }

  async findByDeviceAndDateRange(
    deviceId: number,
    start: Date,
    end: Date,
  ): Promise<Conversation[]> {
    return this.conversationRepository.find({
      where: {
        deviceId,
        startedAt: Between(start, end),
        endedAt: Not(IsNull()),
      },
      order: { startedAt: 'ASC' },
    });
  }
}
```

```typescript
// backend/src/modules/story-card/story-card.controller.ts
import { Controller, Get, Param, UseGuards } from '@nestjs/common';
import { JwtAuthGuard } from '../auth/jwt-auth.guard';
import { CurrentUser } from '../auth/current-user.decorator';
import { JwtPayload } from '../auth/auth.service';
import { StoryCardService } from './story-card.service';
import { DeviceService } from '../device/device.service';
import { CustomHttpException } from '../../common/response/custom-http.exception';
import { ErrorCode } from '../../common/response/error-code.enum';
import { HttpStatus } from '@nestjs/common';

@Controller('story-cards')
@UseGuards(JwtAuthGuard)
export class StoryCardController {
  constructor(
    private readonly storyCardService: StoryCardService,
    private readonly deviceService: DeviceService,
  ) {}

  @Get(':devicePid')
  async findByDevice(
    @CurrentUser() user: JwtPayload,
    @Param('devicePid') devicePid: string,
  ) {
    // Verify the device is linked to this user
    const devices = await this.deviceService.getLinkedDevices(user.sub);
    const device = devices.find((d) => d.pid === devicePid);
    if (!device) {
      throw new CustomHttpException(
        HttpStatus.FORBIDDEN,
        ErrorCode.FORBIDDEN,
        'This device is not linked to your account',
      );
    }

    const cards = await this.storyCardService.findByDeviceId(device.id);
    return {
      cards: cards.map((c) => ({
        pid: c.pid,
        type: c.type,
        cardDate: c.cardDate,
        data: c.data,
        createdAt: c.createdAt,
      })),
    };
  }
}
```

```typescript
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { GoogleGenAI, Session } from '@google/genai';

export interface VoiceSession {
  session: Session;
  transcriptParts: string[];
}

@Injectable()
export class GeminiLiveService {
  private readonly logger = new Logger(GeminiLiveService.name);
  private readonly ai: GoogleGenAI;
  private readonly model: string;

  constructor(private readonly configService: ConfigService) {
    this.ai = new GoogleGenAI({
      apiKey: this.configService.getOrThrow<string>('GEMINI_API_KEY'),
    });
    this.model = this.configService.get<string>(
      'GEMINI_MODEL',
      'gemini-2.0-flash-live-001',
    );
  }

  async createSession(
    onAudio: (audio: string) => void,
    onText: (text: string) => void,
    onInterrupted: () => void,
    onEnd: () => void,
  ): Promise<VoiceSession> {
    const transcriptParts: string[] = [];

    const session = await this.ai.live.connect({
      model: this.model,
      config: {
        responseModalities: ['AUDIO'],
        systemInstruction: {
          parts: [{ text: this.getSoulPrompt() }],
        },
        speechConfig: {
          voiceConfig: {
            prebuiltVoiceConfig: { voiceName: 'Kore' },
          },
        },
      },
      callbacks: {
        onAudio: (data) => {
          onAudio(data.data);
        },
        onText: (text) => {
          transcriptParts.push(`AI: ${text}`);
          onText(text);
        },
        onInterrupted: () => {
          onInterrupted();
        },
        onError: (e) => {
          this.logger.error('Gemini session error', e);
        },
        onClose: () => {
          onEnd();
        },
      },
    });

    return { session, transcriptParts };
  }

  sendAudio(session: Session, audioBase64: string): void {
    session.sendRealtimeInput({
      media: {
        data: audioBase64,
        mimeType: 'audio/pcm;rate=16000',
      },
    });
  }

  sendText(session: Session, text: string): void {
    session.sendClientContent({
      turns: [{ role: 'user', parts: [{ text }] }],
    });
  }

  async closeSession(session: Session): Promise<void> {
    try {
      session.close();
    } catch (e) {
      this.logger.warn('Error closing Gemini session', e);
    }
  }

  private getSoulPrompt(): string {
    return `당신은 "할마이"라는 이름의 AI 말동무입니다.
한국의 고령 어르신과 대화하는 따뜻한 친구 역할을 합니다.

- 어르신이 하시는 말씀에 진심으로 공감하고 경청합니다
- 짧고 자연스러운 문장으로 말합니다
- 가끔 적절한 질문을 해서 대화를 이어갑니다
- 건강, 날씨, 일상, 추억 등 편안한 주제로 대화합니다
- 어르신이 반복하시거나 잊으시는 것에 대해 절대 지적하지 않습니다
```

```typescript
// backend/src/modules/voice/voice.gateway.ts
import { Logger } from '@nestjs/common';
import {
  OnGatewayConnection,
  OnGatewayDisconnect,
  SubscribeMessage,
  WebSocketGateway,
} from '@nestjs/websockets';
import { WebSocket } from 'ws';
import { GeminiLiveService, VoiceSession } from './gemini-live.service';
import { ConversationService } from '../conversation/conversation.service';
import { InjectRepository } from '@nestjs/typeorm';
import { Device } from '../../common/entity/device.entity';
import { Repository } from 'typeorm';

interface ClientState {
  geminiSession: VoiceSession | null;
  conversationId: number | null;
  deviceId: number | null;
  userTranscriptParts: string[];
}

@WebSocketGateway({ path: '/ws/voice' })
export class VoiceGateway implements OnGatewayConnection, OnGatewayDisconnect {
  private readonly logger = new Logger(VoiceGateway.name);
  private readonly clients = new Map<WebSocket, ClientState>();

  constructor(
    private readonly geminiLiveService: GeminiLiveService,
    private readonly conversationService: ConversationService,
    @InjectRepository(Device)
    private readonly deviceRepository: Repository<Device>,
  ) {}

  handleConnection(client: WebSocket) {
    this.clients.set(client, {
      geminiSession: null,
      conversationId: null,
      deviceId: null,
      userTranscriptParts: [],
    });
    this.logger.log('Client connected');
  }

  async handleDisconnect(client: WebSocket) {
    const state = this.clients.get(client);
    if (state) {
      await this.endSession(client, state);
    }
    this.clients.delete(client);
    this.logger.log('Client disconnected');
  }

  @SubscribeMessage('start')
  async handleStart(client: WebSocket, data: { deviceId: string }) {
    const state = this.clients.get(client);
    if (!state) return;

    // Find device by deviceId string
    const device = await this.deviceRepository.findOneBy({
      deviceId: data.deviceId,
    });
    if (!device) {
      this.send(client, 'error', { message: 'Device not found' });
      return;
    }

    state.deviceId = device.id;

    // Create conversation record
    const conversation = await this.conversationService.create(device.id);
    state.conversationId = conversation.id;

    // Create Gemini Live session
    const voiceSession = await this.geminiLiveService.createSession(
      // onAudio: relay AI audio to client
      (audioBase64) => {
        this.send(client, 'audio', { data: audioBase64 });
      },
      // onText: AI transcript
      (text) => {
        this.send(client, 'transcript', { role: 'ai', text });
      },
      // onInterrupted
      () => {
        this.send(client, 'interrupted', {});
      },
      // onEnd
      () => {
        this.send(client, 'ended', {});
      },
    );

    state.geminiSession = voiceSession;
    this.send(client, 'started', { conversationPid: conversation.pid });
    this.logger.log(`Voice session started for device ${data.deviceId}`);
  }

  @SubscribeMessage('audio')
  handleAudio(client: WebSocket, data: { data: string }) {
    const state = this.clients.get(client);
    if (!state?.geminiSession) return;

    this.geminiLiveService.sendAudio(
      state.geminiSession.session,
      data.data,
    );
  }

  @SubscribeMessage('user_transcript')
  handleUserTranscript(client: WebSocket, data: { text: string }) {
    const state = this.clients.get(client);
    if (!state) return;
    state.userTranscriptParts.push(`User: ${data.text}`);
  }

  @SubscribeMessage('end')
  async handleEnd(client: WebSocket) {
    const state = this.clients.get(client);
    if (!state) return;
    await this.endSession(client, state);
    this.send(client, 'ended', {});
  }

  private async endSession(client: WebSocket, state: ClientState) {
    if (state.geminiSession) {
      await this.geminiLiveService.closeSession(
        state.geminiSession.session,
      );

      // Combine transcripts
      if (state.conversationId) {
        const allParts = this.mergeTranscripts(
          state.userTranscriptParts,
          state.geminiSession.transcriptParts,
        );
        await this.conversationService.end(
          state.conversationId,
          allParts.join('\n'),
        );
      }

      state.geminiSession = null;
      state.conversationId = null;
    }
  }

  private mergeTranscripts(
    userParts: string[],
    aiParts: string[],
  ): string[] {
    // Simple interleave - in practice timestamps would be better
    // For now, just concatenate both arrays
    return [...userParts, ...aiParts];
  }

  private send(client: WebSocket, event: string, data: unknown) {
    if (client.readyState === WebSocket.OPEN) {
      client.send(JSON.stringify({ event, data }));
    }
  }
}
```

**Step 1: Create voice client**

```typescript
// frontend/src/lib/voice-client.ts
const SAMPLE_RATE = 16000;
const BUFFER_SIZE = 4096;

export type VoiceState = 'idle' | 'connecting' | 'listening' | 'speaking' | 'ending';

export interface VoiceClientCallbacks {
  onStateChange: (state: VoiceState) => void;
  onAiTranscript: (text: string) => void;
  onError: (error: string) => void;
}

export class VoiceClient {
  private ws: WebSocket | null = null;
  private audioContext: AudioContext | null = null;
  private mediaStream: MediaStream | null = null;
  private processorNode: ScriptProcessorNode | null = null;
  private playbackQueue: string[] = [];
  private isPlaying = false;
  private callbacks: VoiceClientCallbacks;
  private deviceId: string;

  constructor(deviceId: string, callbacks: VoiceClientCallbacks) {
    this.deviceId = deviceId;
    this.callbacks = callbacks;
  }

  async start(): Promise<void> {
    this.callbacks.onStateChange('connecting');

    try {
      // 1. Get microphone access
      this.mediaStream = await navigator.mediaDevices.getUserMedia({
        audio: {
          sampleRate: SAMPLE_RATE,
          channelCount: 1,
          echoCancellation: true,
          noiseSuppression: true,
        },
      });

      // 2. Set up AudioContext for mic capture
      this.audioContext = new AudioContext({ sampleRate: SAMPLE_RATE });
      const source = this.audioContext.createMediaStreamSource(this.mediaStream);
      this.processorNode = this.audioContext.createScriptProcessor(BUFFER_SIZE, 1, 1);

      this.processorNode.onaudioprocess = (e) => {
        if (this.ws?.readyState === WebSocket.OPEN) {
          const pcmData = e.inputBuffer.getChannelData(0);
          const base64 = this.float32ToBase64PCM16(pcmData);
          this.send('audio', { data: base64 });
        }
      };

      source.connect(this.processorNode);
      this.processorNode.connect(this.audioContext.destination);

      // 3. Connect WebSocket
      const wsUrl =
        (process.env.NEXT_PUBLIC_WS_URL || 'ws://localhost:4000') + '/ws/voice';
      this.ws = new WebSocket(wsUrl);

      this.ws.onopen = () => {
        this.send('start', { deviceId: this.deviceId });
      };

      this.ws.onmessage = (event) => {
        const msg = JSON.parse(event.data);
        this.handleMessage(msg.event, msg.data);
      };

      this.ws.onerror = () => {
        this.callbacks.onError('Connection error');
        this.stop();
      };

      this.ws.onclose = () => {
        if (this.audioContext) {
          this.callbacks.onStateChange('idle');
        }
      };
    } catch (err) {
      this.callbacks.onError(
        err instanceof Error ? err.message : 'Failed to start',
      );
      this.stop();
    }
  }

  stop(): void {
    this.callbacks.onStateChange('ending');

    if (this.ws?.readyState === WebSocket.OPEN) {
      this.send('end', {});
    }

    this.cleanup();
    setTimeout(() => this.callbacks.onStateChange('idle'), 1500);
  }

  private handleMessage(event: string, data: Record<string, unknown>) {
    switch (event) {
      case 'started':
        this.callbacks.onStateChange('listening');
        break;
      case 'audio':
        this.callbacks.onStateChange('speaking');
        this.enqueueAudio(data.data as string);
        break;
      case 'transcript':
        this.callbacks.onAiTranscript(data.text as string);
        break;
      case 'interrupted':
        this.callbacks.onStateChange('listening');
        this.playbackQueue = [];
        break;
      case 'ended':
        this.cleanup();
        this.callbacks.onStateChange('idle');
        break;
      case 'error':
        this.callbacks.onError(data.message as string);
        break;
    }
  }

  private async enqueueAudio(base64: string) {
    this.playbackQueue.push(base64);
    if (!this.isPlaying) {
      await this.playNext();
    }
  }

  private async playNext() {
    if (this.playbackQueue.length === 0 || !this.audioContext) {
      this.isPlaying = false;
      // Done speaking, back to listening
      this.callbacks.onStateChange('listening');
      return;
    }

    this.isPlaying = true;
    const base64 = this.playbackQueue.shift()!;

    try {
      const pcm16 = Uint8Array.from(atob(base64), (c) => c.charCodeAt(0));
      const float32 = this.pcm16ToFloat32(pcm16);
      const buffer = this.audioContext.createBuffer(1, float32.length, SAMPLE_RATE);
      buffer.getChannelData(0).set(float32);

      const source = this.audioContext.createBufferSource();
      source.buffer = buffer;
      source.connect(this.audioContext.destination);
      source.onended = () => this.playNext();
      source.start();
    } catch {
      this.playNext();
    }
  }

  private cleanup() {
    this.processorNode?.disconnect();
    this.processorNode = null;

    if (this.mediaStream) {
      this.mediaStream.getTracks().forEach((t) => t.stop());
      this.mediaStream = null;
    }

    if (this.audioContext?.state !== 'closed') {
      this.audioContext?.close();
    }
    this.audioContext = null;

    if (this.ws) {
      this.ws.onclose = null;
      if (this.ws.readyState === WebSocket.OPEN) {
        this.ws.close();
      }
      this.ws = null;
    }

    this.playbackQueue = [];
    this.isPlaying = false;
  }

  private send(event: string, data: unknown) {
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify({ event, data }));
    }
  }

  private float32ToBase64PCM16(float32: Float32Array): string {
    const pcm16 = new Int16Array(float32.length);
    for (let i = 0; i < float32.length; i++) {
      const s = Math.max(-1, Math.min(1, float32[i]));
      pcm16[i] = s < 0 ? s * 0x8000 : s * 0x7fff;
    }
    const bytes = new Uint8Array(pcm16.buffer);
    let binary = '';
    for (let i = 0; i < bytes.length; i++) {
      binary += String.fromCharCode(bytes[i]);
    }
    return btoa(binary);
  }

  private pcm16ToFloat32(pcm16: Uint8Array): Float32Array {
    const int16 = new Int16Array(pcm16.buffer);
    const float32 = new Float32Array(int16.length);
    for (let i = 0; i < int16.length; i++) {
      float32[i] = int16[i] / 0x8000;
    }
    return float32;
  }
}
```

```typescript
@Injectable()
export class CardGeneratorService {
  private readonly logger = new Logger(CardGeneratorService.name);
  private readonly ai: GoogleGenAI;

  constructor(
    @InjectRepository(Device)
    private readonly deviceRepository: Repository<Device>,
    private readonly conversationService: ConversationService,
    private readonly storyCardService: StoryCardService,
    private readonly configService: ConfigService,
  ) {

      apiKey: this.configService.getOrThrow<string>('GEMINI_API_KEY'),
    });

  @Cron('0 10 * * *', { timeZone: 'Asia/Seoul' })
  async generateDailyCards() {
    this.logger.log('Starting daily story card generation');

    const yesterday = subDays(new Date(), 1);
    const dateStr = format(yesterday, 'yyyy-MM-dd');
    const dayStart = startOfDay(yesterday);
    const dayEnd = endOfDay(yesterday);

    // Find all active devices (linked)
    const devices = await this.deviceRepository.find({
      where: { linkedAt: Not(IsNull()) },
    });

    for (const device of devices) {
      try {
        await this.generateCardForDevice(device.id, dateStr, dayStart, dayEnd);
      } catch (e) {
        this.logger.error(
          `Failed to generate card for device ${device.id}`,
          e,
        );
      }
    }

    this.logger.log(`Daily card generation complete. Processed ${devices.length} devices.`);
  }

  private async generateCardForDevice(
    deviceId: number,
    dateStr: string,
    dayStart: Date,
    dayEnd: Date,
  ) {
    // Skip if card already exists
    if (await this.storyCardService.existsForDate(deviceId, dateStr)) {
      return;
    }

    // Get all completed conversations from yesterday
    const conversations =
      await this.conversationService.findByDeviceAndDateRange(
        deviceId,
        dayStart,
        dayEnd,
      );

    if (conversations.length === 0) {
      return; // No conversations, no card
    }

    // Combine transcripts
    const combinedTranscript = conversations
      .map((c) => c.transcript || '')
      .filter(Boolean)
      .join('\n---\n');

    if (!combinedTranscript.trim()) {
      return;
    }

    // Generate card data with Gemini
    const cardData = await this.generateCardData(combinedTranscript);
    if (!cardData) return;

    await this.storyCardService.create(deviceId, dateStr, cardData);
    this.logger.log(`Card generated for device ${deviceId}, date ${dateStr}`);
  }

3. vibe: 대화의 전체적인 분위기를 하나 선택 (warm / calm / quiet)

반드시 아래 JSON 형식으로만 응답하세요. 다른 텍스트는 포함하지 마세요.
{"topic": "...", "quote": "...", "vibe": "warm|calm|quiet"}

## 대화 기록
${transcript}`;

    try {
      const response = await this.ai.models.generateContent({
        model: 'gemini-2.0-flash',
        contents: prompt,
      });

      const text = response.text?.trim();
      if (!text) return null;
```
